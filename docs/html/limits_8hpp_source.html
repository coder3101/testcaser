136  void add_interval_exception(std::pair<T, T> interval) {
137  T upper =
138  interval.first > interval.second ? interval.first : interval.second;
139  T lower = interval.first + interval.second - upper;

158  bool valid_output(T out) const {
159  if (out >= UpperLimit || out < LowerLimit) return false;
160  for (auto& e : except_intervals)
161  if (out >= e.lower && out < e.upper) return false;
162  return true;

169  unsigned long long actual_limit_size() {
170  T min, max;
171  unsigned long long total_size = UpperLimit - LowerLimit;
172  if (except_intervals.size() == 0) return total_size;
173  std::sort(except_intervals.begin(), except_intervals.end());
174  min = except_intervals[0].lower;
175  max = except_intervals[except_intervals.size() - 1].upper;
176  total_size -= max - min;
177  auto active = except_intervals[0];
178  for (int t = 1; t < except_intervals.size(); t++) {
179  if (except_intervals[t].lower >= active.upper)
180  total_size += except_intervals[t].lower - active.upper;
181  if (active.upper < except_intervals[t].upper)
182  active = except_intervals[t];
183  }
184  return total_size;
185  };
