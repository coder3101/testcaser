171  if (except_intervals.size() == 0) return total_size;
172  std::sort(except_intervals.begin(), except_intervals.end());
173  min = except_intervals[0].lower;
174  max = except_intervals[except_intervals.size() - 1].upper;
175  total_size -= max - min;
176  auto active = except_intervals[0];

178  if (except_intervals[t].lower >= active.upper)
179  total_size += except_intervals[t].lower - active.upper;
180  if (active.upper < except_intervals[t].upper)
181  active = except_intervals[t];
